SELECT I.SUMMARY, IL.LIST_NAME
FROM ISSUES I
JOIN ISSUES_LIST IL ON I.ISSUE_LIST_ID = IL.ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNED_TO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNED_TO
GROUP BY I.USER_ID_ASSIGNED_TO;
